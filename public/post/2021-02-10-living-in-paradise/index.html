Build a table

data(mtcars)
mtcars$model <- row.names(mtcars)
rownames(mtcars) <- NULL
dplyr::filter(mtcars, cyl == 4) %>%
  select(model, mpg, cyl) %>%
  arrange(-mpg) %>%
  kableExtra::kbl()

Make a plot

mtcars$cyl <- factor(mtcars$cyl)
p <- ggplot(mtcars, aes(mpg, wt, group = cyl, color = cyl))
p <- p + geom_point()
p <- p + ggtitle("Mtcars")
